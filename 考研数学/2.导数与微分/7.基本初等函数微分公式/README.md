## 常数

$$
d(C) = 0
$$

## 三角函数

$$
d(sinx) = {cosx}dx
\\
d(cosx) = {-sinx}dx
\\
d(tanx) = {(secx) ^ 2}dx
\\
d(cotx) = {(cscx) ^ 2}dx
\\
d(secx) = {secxtanx}dx
\\
d(cscx) = {-cscxcotx}dx
$$

## 幂函数

$$
d(x ^ n) = {nx ^ {n - 1}}dx
$$

## 指数函数

$$
d(a ^ x) = {a ^ x \ln ^ {a}}dx
\\
d(e ^ x) = {e ^ x}dx
\\
$$

## 对数函数

$$
d(\log_a^{x}) = {\frac{1}{x\ln^a}}dx
\\
d(\ln^{|x|}) = {\frac{1}{x}}dx
$$

## 反三角函数

$$
d(arcsinx) = {\frac{1}{\sqrt{1 - x ^ 2}}}dx
\\
d(arccosx) = {-\frac{1}{\sqrt{1 - x ^ 2}}}dx
\\
d(arctanx) = {\frac{1}{1 + x ^ 2}}dx
\\
d(arccotx) = {-\frac{1}{1 + x ^ 2}}dx
$$
